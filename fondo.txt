import 'dart:ui'; // ⚠️ Súper importante importar esto para el ImageFilter
import 'package:flutter/material.dart';

Stack(
  fit: StackFit.expand,
  children: [
    // 1. Tu Imagen de Fondo
    Image.asset(

      fit: BoxFit.cover,
    ),
    
    // 2. La Tarjeta de Vidrio Esmerilado (Glassmorphism)
    Center(

        child: BackdropFilter(
          // Aquí controlas qué tan borroso es el fondo (10 a 15 es ideal)

              // El borde sutil le da el efecto de reflejo en la orilla del vidrio
              border: Border.all(
                color: Colors.white.withOpacity(0.2),
                width: 1.5,
              ),
              borderRadius: BorderRadius.circular(20.0),
            ),
            child: const Text(

              textAlign: TextAlign.center,
              style: TextStyle(
                color: Colors.white,
                fontSize: 48,
                fontWeight: FontWeight.bold,
                // Le dejamos una sombrita muy leve al texto por si acaso
                shadows: [
                  Shadow(
                    offset: Offset(1.0, 1.0),
                    blurRadius: 3.0,
                    color: Colors.black54,
                  ),
